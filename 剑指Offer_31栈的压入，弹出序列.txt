class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        int n1 = pushed.size();
        int n2 = popped.size();
        stack<int> st;
        for(int i = 0,j = 0; i < n1; i++)
        {
            st.push(pushed[i]);
            while(!st.empty() && st.top() == popped[j]){
                j++;
                st.pop();
            }
        }
        return st.empty();
    }
};

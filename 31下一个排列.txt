Solution1 感觉应该是固定解法，很神奇，从右往左寻找一个nums[i] < nums[i+1]的数，然后再从右往左找一个nuns[j] > nums[i]
的数，交换，然后将i往右的数逆序排列
class Solution {
public:
    void nextPermutation(vector<int>& nums) {
        int i = nums.size() - 2;
        while(i >= 0 && nums[i] >= nums[i + 1])i--;

        if(i >= 0)
        {
            int j = nums.size() - 1;
            while(j >=0 && nums[i] >= nums[j])j--;

            swap(nums[i] , nums[j]);
        }
        reverse(nums.begin() + i + 1, nums.end());
    }
};

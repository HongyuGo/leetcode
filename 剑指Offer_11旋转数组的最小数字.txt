class Solution {
public:
    int minArray(vector<int>& numbers) {
        int n = numbers.size();
        int low = 0, high = n - 1;
        int mid = 0;
        while(low < high)
        {
           mid = low + (high - low) / 2;
           if(numbers[mid] < numbers[high]) 
                high = mid;
            else if(numbers[mid] > numbers[high])           
                low = mid+1;
            else
                high = high - 1;
        }
        return numbers[high];
    }
};

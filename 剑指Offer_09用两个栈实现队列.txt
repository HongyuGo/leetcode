class CQueue {
public:
    CQueue() {

    }
    
    void appendTail(int value) {
        instack.push(value);
    }
    
    int deleteHead() {
        if(outstack.empty())
        {
            if(instack.empty())
                return -1;
            else
                while(!instack.empty())
                {
                    outstack.push(instack.top());
                    instack.pop();    
                }
        }
        int data = outstack.top();
        outstack.pop();
        return data;
    }
private:
    stack<int> instack;
    stack<int> outstack;

};

class Solution {
public:
    int fib(int n) {
        if(n == 0) return 0;
        if(n == 1) return 1;
        int a_last = 0,a_now = 1;
        int temp  = 0;
        for(int i = 2; i<=n ; i++)
        {
            temp = a_now + a_last;
            temp = temp % 1000000007;
            a_last = a_now;
            a_now = temp;
        }
        return temp ;
    }
};

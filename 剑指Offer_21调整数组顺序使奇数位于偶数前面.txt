class Solution {
public:
    vector<int> exchange(vector<int>& nums) {
        int i = 0, j = nums.size()-1; 
        for(;i<j;)
        {
            if(nums[i] % 2 == 0)
                {
                    if(nums[j] % 2 == 0)
                        j--;
                    else
                        swap(nums[i],nums[j]),j--,i++;
                }
            else
                i++;
        }
        return nums;
    }
};

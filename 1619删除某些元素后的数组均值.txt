class Solution {
public:
    double trimMean(vector<int>& arr) {
        sort(arr.begin(),arr.end());
        int n = arr.size();
        int x = n * 0.05;
        double sum = 0; 
        for(int i = x; i<n-x; i++)
        {
            sum = sum + arr[i];
        }
        double average = sum / (n - 2 * x);
        return average; 

    }
};
